# Benchmark Command Reference
# --------------------------

# Basic usage with default parameters
uv run main.py

# Process files from a specific directory (3 runs)
uv run main.py -fx_path ./data/fx -runs 3

# Limit to specific number of files (note: use -files NOT --files)
uv run main.py -fx_path ./data/fx -runs 3 -files 35

# Running specific benchmark sizes -- 35=1gb 160=5gb 320=10gb default=31gb
uv run main.py -files 35 -runs 3 
uv run main.py -files 160 -runs 3
uv run main.py -files 320 -runs 3
uv run main.py -runs 3
